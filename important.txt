  <script type="module">
    import { getFirestore, doc, setDoc, serverTimestamp, updateDoc, increment } from "https://www.gstatic.com/firebasejs/9.6.10/firebase-firestore.js";
    import { getAuth, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/9.6.10/firebase-auth.js";

    const ARTICLE_ID = "my-new-article"; // ‚úÖ Change for each article
    const ARTICLE_TITLE = "My New Article"; // ‚úÖ Human-friendly name

    const auth = getAuth();
    const db = getFirestore();
    let hasScrolledToBottom = false;

    onAuthStateChanged(auth, (user) => {
      if (!user) return;

      window.addEventListener('scroll', async () => {
        if (hasScrolledToBottom) return;

        const scrollTop = window.scrollY;
        const docHeight = document.documentElement.scrollHeight;
        const winHeight = window.innerHeight;
        const scrollPercent = (scrollTop + winHeight) / docHeight;

        if (scrollPercent > 0.9) {
          hasScrolledToBottom = true;

          // Log user read
          await setDoc(doc(db, "reads", `${user.uid}-${ARTICLE_ID}`), {
            userId: user.uid,
            articleId: ARTICLE_ID,
            title: ARTICLE_TITLE,
            timestamp: serverTimestamp(),
            duration: 300 // Optional: Estimated read time in seconds
          });

          // Update global article read count
          const articleRef = doc(db, "articles", ARTICLE_ID);
          await setDoc(articleRef, { readCount: increment(1) }, { merge: true });
        }
      });
    });
  </script>

  <script type="module">
    // Firebase setup
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";
    import { getFirestore, collection, addDoc } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";

    // Star rating logic
    let selectedRating = 0;
    const stars = document.querySelectorAll('.star');

    stars.forEach((star) => {
      star.addEventListener('mouseover', () => {
        const val = parseInt(star.getAttribute('data-value'));
        stars.forEach(s => {
          const sVal = parseInt(s.getAttribute('data-value'));
          s.style.color = sVal <= val ? 'gold' : 'lightgray';
        });
      });

      star.addEventListener('mouseleave', () => {
        stars.forEach(s => {
          const sVal = parseInt(s.getAttribute('data-value'));
          s.style.color = sVal <= selectedRating ? 'gold' : 'lightgray';
        });
      });

      star.addEventListener('click', () => {
        selectedRating = parseInt(star.getAttribute('data-value'));
        stars.forEach(s => {
          const sVal = parseInt(s.getAttribute('data-value'));
          s.style.color = sVal <= selectedRating ? 'gold' : 'lightgray';
        });
      });
    });

    document.getElementById('submit-rating').addEventListener('click', async () => {
      if (selectedRating === 0) {
        alert("Please select a rating before submitting.");
        return;
      }

      try {
        await addDoc(collection(db, "ratings"), {
          rating: selectedRating,
          timestamp: new Date()
        });

        document.getElementById("feedback").style.display = "block";
      } catch (e) {
        console.error("Error adding document: ", e);
        alert("Something went wrong. Check the console for details.");
      }
    });
  </script>
